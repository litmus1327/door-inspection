Configuration
Connect to Supabase for multi-inspector cloud sync.
Supabase Connection
1. Go to supabase.com → create a free project.
2. In your project: SQL Editor → run the setup SQL below.
3. Go to Settings → Data API → copy your Project URL and anon/public key.
4. Paste them here and click Save.
▶ VIEW SUPABASE SETUP SQL 
CREATE TABLE door_inspections (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  icon_no text,
  asset_id text,
  floor text,
  grid_block text,
  assembly_type text,
  door_rating text,
  inspector text,
  project text,
  inspection_date timestamptz DEFAULT now(),
  overall_status text,
  deficiencies jsonb,
  findings jsonb,
  additional_comments text,
  post_inspection_status text,
  synced_at timestamptz DEFAULT now()
);

CREATE TABLE door_schedule (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  asset_id text UNIQUE,
  icon_no text,
  floor text,
  grid_block text,
  assembly_type text,
  door_rating text,
  project text,
  uploaded_at timestamptz DEFAULT now()
);

ALTER TABLE door_inspections ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow all" ON door_inspections FOR ALL USING (true);
ALTER TABLE door_schedule ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow all" ON door_schedule FOR ALL USING (true);
